    81	- (void)connect {
    82	    [_lock lock];
    83	    if (self.status == IdeviceStatusConnected || self.status == IdeviceStatusConnecting) { [_lock unlock]; return; }
    84	    self.status = IdeviceStatusConnecting; self.lastError = nil;
    85	    [_lock unlock];
    86	    dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{ [self _performConnect]; });

    89	- (void)_performConnect {
    90	    NSString *ip = self.ipAddress; uint16_t port = self.port; NSString *pairingPath = self.pairingFilePath;
    91	    struct sockaddr_in sa; memset(&sa, 0, sizeof(sa)); sa.sin_family = AF_INET; sa.sin_port = htons(port);
    92	    if (!ip || ip.length == 0 || inet_pton(AF_INET, [ip UTF8String], &sa.sin_addr) <= 0) { [self _handleError:@"IPアドレスの形式が正しくありません"]; return; }
    93	    struct IdeviceFfiError *err = NULL; struct IdeviceProviderHandle *localProvider = NULL;
    94	    struct LockdowndClientHandle *localLockdown = NULL; struct HeartbeatClientHandle *localHb = NULL;
    95	    struct IdevicePairingFile *pairingForProvider = NULL; struct IdevicePairingFile *pairingForSession = NULL;
    96	    if (pairingPath && pairingPath.length > 0) {
    97	        err = idevice_pairing_file_read([pairingPath UTF8String], &pairingForProvider);
    98	        if (!err) {
    99	            err = idevice_pairing_file_read([pairingPath UTF8String], &pairingForSession);
   100	        }
   101	        if (err || !pairingForProvider || !pairingForSession) {
   102	            [self _handleFfiError:err fallback:@"ペアリングファイルの読み込みに失敗しました"];
   103	            if (pairingForProvider) idevice_pairing_file_free(pairingForProvider);
   104	            if (pairingForSession) idevice_pairing_file_free(pairingForSession);
   105	            return;
   106	        }
   107	    } else {
   108	        [self _handleError:@"ペアリングファイルが選択されていません。設定から選択してください。"];
   109	        return;
   110	    }
   111	    err = idevice_tcp_provider_new((const idevice_sockaddr *)&sa, pairingForProvider, "frappe-idevice", &localProvider);
   112	    if (err || !localProvider) {
   113	        [self _handleFfiError:err fallback:@"プロバイダーの作成に失敗しました"];
   114	        idevice_pairing_file_free(pairingForProvider);
   115	        idevice_pairing_file_free(pairingForSession);
   116	        return;
   117	    }
   118	    // Note: localProvider now owns pairingForProvider logic internally usually,
   119	    // but if idevice_tcp_provider_new fails, we must free it.
   120
   121	    err = lockdownd_connect(localProvider, &localLockdown);
   122	    if (err || !localLockdown) {
   123	        [self _handleFfiError:err fallback:@"Lockdownサービスへの接続に失敗しました"];
   124	        idevice_provider_free(localProvider);
   125	        idevice_pairing_file_free(pairingForSession);
   126	        return;
   127	    }
   128
   129	    err = lockdownd_start_session(localLockdown, pairingForSession);
   130	    if (err) {
   131	        [self _handleFfiError:err fallback:@"セッションの開始に失敗しました"];
   132	        idevice_pairing_file_free(pairingForSession);
   133	        lockdownd_client_free(localLockdown);
   134	        idevice_provider_free(localProvider);
   135	        return;
   136	    }
   137	    err = heartbeat_connect(localProvider, &localHb); if (err) idevice_error_free(err);
   138	    BOOL ddi = NO; struct ImageMounterHandle *mounter = NULL;
   139	    err = image_mounter_connect(localProvider, &mounter);
   140	    if (!err && mounter) {
   141	        plist_t *devices = NULL; size_t count = 0;
   142	        err = image_mounter_copy_devices(mounter, &devices, &count);
   143	        if (!err) {
   144	            ddi = (count > 0);
   145	            if (devices) idevice_plist_array_free(devices, (uintptr_t)count);
   146	        } else {
   147	            if (err) idevice_error_free(err);
   148	        }
   149	        image_mounter_free(mounter);
   150	    } else if (err) idevice_error_free(err);
   151	    [_lock lock];
   152	    if (self.status == IdeviceStatusConnecting) {
   153	        _pairingFile = pairingForSession; _provider = localProvider; _lockdownClient = localLockdown;
   154	        _heartbeatClient = localHb; _heartbeatActive = (localHb != NULL); _ddiMounted = ddi; _status = IdeviceStatusConnected;
   155	        if (localHb) { dispatch_async(dispatch_get_main_queue(), ^{ [self _startHeartbeatTimer]; }); }
   156	        dispatch_async(dispatch_get_main_queue(), ^{ [[NSNotificationCenter defaultCenter] postNotificationName:@"IdeviceStatusChanged" object:nil]; });
   157	    } else {
   158	        if (localHb) heartbeat_client_free(localHb);
   159	        lockdownd_client_free(localLockdown); idevice_provider_free(localProvider); idevice_pairing_file_free(pairingForSession);
   160	    }
   161	    [_lock unlock];
   162	}

   169	- (void)_sendHeartbeat {
   170	    __weak typeof(self) weakSelf = self;
   171	    dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
   172	        __strong typeof(weakSelf) strongSelf = weakSelf; if (!strongSelf) return;
   173	        struct IdeviceFfiError *err = NULL;
   174	        [strongSelf->_lock lock]; if (strongSelf.heartbeatClient) err = heartbeat_send_polo(strongSelf.heartbeatClient); [strongSelf->_lock unlock];
   175	        if (err) { if (err) idevice_error_free(err); [strongSelf disconnect]; strongSelf.lastError = @"ハートビートが途切れました"; }
   176	    });

   183	    if (self.lockdownClient) { lockdownd_client_free(self.lockdownClient); _lockdownClient = NULL; }
   184	    if (self.provider) { idevice_provider_free(self.provider); _provider = NULL; }
   185	    if (self.pairingFile) { idevice_pairing_file_free(self.pairingFile); _pairingFile = NULL; }
   186	    _status = IdeviceStatusDisconnected; _heartbeatActive = NO; _ddiMounted = NO;
   187	    [_lock unlock];
   188	    dispatch_async(dispatch_get_main_queue(), ^{ [[NSNotificationCenter defaultCenter] postNotificationName:@"IdeviceStatusChanged" object:nil]; });

   202	    [_lock lock];
   203	    if (self.status != IdeviceStatusConnected || !self.provider) {
   204	        [_lock unlock];
   205	        if (completion) completion(nil, [NSError errorWithDomain:@"Idevice" code:1 userInfo:@{NSLocalizedDescriptionKey: @"デバイスに接続されていません"}]);

   207	    }

   209	    [_lock unlock];
   210	    dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
   211	        struct InstallationProxyClientHandle *client = NULL;
   212	        struct IdeviceFfiError *err = installation_proxy_connect(p, &client);

   214	            NSString *msg = err ? [NSString stringWithUTF8String:err->message ?: "InstProxyへの接続に失敗しました"] : @"InstProxyへの接続に失敗しました";
   215	            if (err) idevice_error_free(err);
   216	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil, [NSError errorWithDomain:@"Idevice" code:2 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   217	            return;
   218	        }
   219	        void *out_result = NULL; size_t out_result_len = 0;
   220	        err = installation_proxy_get_apps(client, "Any", NULL, 0, &out_result, &out_result_len);
   221	        installation_proxy_client_free(client);
   222	        if (err) {
   223	            NSString *msg = [NSString stringWithUTF8String:err->message ?: "アプリ一覧の取得に失敗しました"];
   224	            if (err) idevice_error_free(err);
   225	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil, [NSError errorWithDomain:@"Idevice" code:3 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   226	            return;
   227	        }
   228	        NSMutableArray *finalApps = [NSMutableArray array];
   229	        if (out_result && out_result_len > 0) {
   230	            plist_t *handles = (plist_t *)out_result;
   231	            for (size_t i = 0; i < out_result_len; i++) {
   232	                if (handles[i]) {
   233	                    id obj = [self _convertPlistToObjC:handles[i] depth:0];
   234	                    if (obj) [finalApps addObject:obj];
   235	                }
   236	            }
   237	            idevice_plist_array_free(handles, out_result_len);
   238	        }
   239	        if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(finalApps, nil); });
   240	    });
   241	}

   243	- (id)_convertPlistToObjC:(plist_t)node depth:(int)depth {
   244	    if (!node || depth > 20) return nil;
   245	    plist_type type = PLIST_NONE;
   246	    type = plist_get_node_type(node);
   247	    switch (type) {
   248	        case PLIST_BOOLEAN: { uint8_t val = 0; plist_get_bool_val(node, &val); return @((BOOL)val); }
   249	        case PLIST_INT: { uint64_t val = 0; plist_get_uint_val(node, &val); return @(val); }
   250	        case PLIST_REAL: { double val = 0; plist_get_real_val(node, &val); return @(val); }
   251	        case PLIST_STRING: { char *val = NULL; plist_get_string_val(node, &val); NSString *s = (val) ? [NSString stringWithUTF8String:val] : @""; if (val) plist_mem_free(val); return s; }
   252	        case PLIST_KEY: { char *val = NULL; plist_get_key_val(node, &val); NSString *s = (val) ? [NSString stringWithUTF8String:val] : @""; if (val) plist_mem_free(val); return s; }
   253	        case PLIST_ARRAY: {
   254	            uint32_t size = plist_array_get_size(node);
   255	            if (size > 1000) size = 1000;
   256	            NSMutableArray *arr = [NSMutableArray arrayWithCapacity:size];
   257	            for (uint32_t i = 0; i < size; i++) {
   258	                id obj = [self _convertPlistToObjC:plist_array_get_item(node, i) depth:depth + 1];
   259	                if (obj) [arr addObject:obj];
   260	            }
   261	            return arr;
   262	        }
   263	        case PLIST_DICT: {
   264	            uint32_t size = plist_dict_get_size(node);
   265	            if (size > 1000) size = 1000;
   266	            NSMutableDictionary *dict = [NSMutableDictionary dictionaryWithCapacity:size];
   267	            plist_dict_iter iter = NULL;
   268	            plist_dict_new_iter(node, &iter);
   269	            if (iter) {
   270	                char *key = NULL; plist_t subnode = NULL;
   271	                for (uint32_t i = 0; i < size; i++) {
   272	                    plist_dict_next_item(node, iter, &key, &subnode);
   273	                    if (key) {
   274	                        NSString *nsKey = [NSString stringWithUTF8String:key];
   275	                        id obj = [self _convertPlistToObjC:subnode depth:depth + 1];
   276	                        if (nsKey && obj) dict[nsKey] = obj;
   277	                        plist_mem_free(key);
   278	                        key = NULL;
   279	                    } else {
   280	                        break;
   281	                    }
   282	                }
   283	                free(iter);
   284	            }
   285	            return dict;
   286	        }
   287	        case PLIST_DATA: { uint64_t len = 0; const char *ptr = plist_get_data_ptr(node, &len); return (ptr && len > 0 && len < 5*1024*1024) ? [NSData dataWithBytes:ptr length:(NSUInteger)len] : [NSData data]; }
   288	        default: return nil;
   289	    }
   290	}

   292	- (void)launchAppWithBundleId:(NSString *)bundleId completion:(void (^)(NSError *error))completion {
   293	    [_lock lock];
   294	    if (self.status != IdeviceStatusConnected || !self.provider || !self.lockdownClient) {
   295	        [_lock unlock];
   296	        if (completion) completion([NSError errorWithDomain:@"Idevice" code:1 userInfo:@{NSLocalizedDescriptionKey: @"デバイスに接続されていません"}]);
   297	        return;
   298	    }
   299	    struct LockdowndClientHandle *l = self.lockdownClient;
   300	    [_lock unlock];
   301
   302	    dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
   303	        uint16_t service_port = 0;
   304	        bool ssl = false;
   305	        struct IdeviceFfiError *err = lockdownd_start_service(l, "com.apple.instruments.server.services.deviceinfo", &service_port, &ssl);
   306	        if (err) {
   307	            if (err) idevice_error_free(err);
   308	            err = lockdownd_start_service(l, "com.apple.mobile.installation_proxy", &service_port, &ssl);
   309	        }
   310
   311	        if (err) {
   312	            NSString *msg = [NSString stringWithUTF8String:err->message ?: "サービスの開始に失敗しました"];
   313	            if (err) idevice_error_free(err);
   314	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion([NSError errorWithDomain:@"Idevice" code:4 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   315	            return;
   316	        }
   317
   318	        [[Logger sharedLogger] log:[NSString stringWithFormat:@"[Idevice] Successfully started a service on port %d for launch attempt of %@", service_port, bundleId]];
   319	        if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil); });
   320	    });
   321	}

   324	- (void)getRsdServicesWithCompletion:(void (^)(NSArray *services, NSError *error))completion {
   325	    [_lock lock];
   326	    if (self.status != IdeviceStatusConnected || !self.provider) {
   327	        [_lock unlock];
   328	        if (completion) completion(nil, [NSError errorWithDomain:@"Idevice" code:1 userInfo:@{NSLocalizedDescriptionKey: @"デバイスに接続されていません"}]);
   329	        return;
   330	    }
   331	    struct LockdowndClientHandle *lockdown = self.lockdownClient;
   332	    NSString *ipStr = self.ipAddress;
   333	    [_lock unlock];
   334
   335	    dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
   336	        uint16_t rsd_port = 0;
   337	        bool ssl = false;
   338	        struct IdeviceFfiError *err = lockdownd_start_service(lockdown, "com.apple.mobile.restored", &rsd_port, &ssl);
   339	        if (err) {
   340	            NSString *msg = [NSString stringWithUTF8String:err->message ?: "RSDサービスの開始に失敗しました"];
   341	            if (err) idevice_error_free(err);
   342	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil, [NSError errorWithDomain:@"Idevice" code:6 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   343	            return;
   344	        }
   345
   346	        struct TcpFeedObject *feeder = NULL;
   347	        struct TcpEatObject *eater = NULL;
   348	        struct AdapterHandle *adapter = NULL;
   349	        err = idevice_tcp_stack_into_sync_objects("0.0.0.0", [ipStr UTF8String], &feeder, &eater, &adapter);
   350	        if (err || !adapter) {
   351	            NSString *msg = err ? [NSString stringWithUTF8String:err->message ?: "TCPスタックの初期化に失敗しました"] : @"TCPスタックの初期化に失敗しました";
   352	            if (err) idevice_error_free(err);
   353	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil, [NSError errorWithDomain:@"Idevice" code:7 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   354	            return;
   355	        }
   356
   357	        struct ReadWriteOpaque *stream = NULL;
   358	        err = adapter_connect(adapter, rsd_port, &stream);
   359	        if (err || !stream) {
   360	            NSString *msg = err ? [NSString stringWithUTF8String:err->message ?: "RSDストリームの接続に失敗しました"] : @"RSDストリーム de 接続に失敗しました";
   361	            if (err) idevice_error_free(err);
   362	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil, [NSError errorWithDomain:@"Idevice" code:8 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   363	            return;
   364	        }
   365
   366	        struct RsdHandshakeHandle *handshake = NULL;
   367	        err = rsd_handshake_new(stream, &handshake);
   368	        if (err || !handshake) {
   369	            NSString *msg = err ? [NSString stringWithUTF8String:err->message ?: "RSDハンドシェイクに失敗しました"] : @"RSDハンドシェイクに失敗しました";
   370	            if (err) idevice_error_free(err);
   371	            idevice_stream_free(stream);
   372	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil, [NSError errorWithDomain:@"Idevice" code:9 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   373	            return;
   374	        }
   375
   376	        struct CRsdServiceArray *rawServices = NULL;
   377	        err = rsd_get_services(handshake, &rawServices);
   378	        if (err || !rawServices) {
   379	            NSString *msg = err ? [NSString stringWithUTF8String:err->message ?: "RSDサービスの取得に失敗しました"] : @"RSDサービスの取得に失敗しました";
   380	            if (err) idevice_error_free(err);
   381	            rsd_handshake_free(handshake);
   382	            idevice_stream_free(stream);
   383	            if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(nil, [NSError errorWithDomain:@"Idevice" code:10 userInfo:@{NSLocalizedDescriptionKey: msg}]); });
   384	            return;
   385	        }
   386
   387	        NSMutableArray *results = [NSMutableArray array];
   388	        for (size_t i = 0; i < rawServices->count; i++) {
   389	            struct CRsdService *s = &rawServices->services[i];
   390	            NSMutableDictionary *dict = [NSMutableDictionary dictionary];
   391	            if (s->name) dict[@"name"] = [NSString stringWithUTF8String:s->name];
   392	            if (s->entitlement) dict[@"entitlement"] = [NSString stringWithUTF8String:s->entitlement];
   393	            dict[@"port"] = @(s->port);
   394	            [results addObject:dict];
   395	        }
   396
   397	        rsd_free_services(rawServices);
   398	        rsd_handshake_free(handshake);
   399	        idevice_stream_free(stream);
   400
   401	        if (completion) dispatch_async(dispatch_get_main_queue(), ^{ completion(results, nil); });
   402	    });
   403	}
   404	- (void)captureSysdiagnoseWithCompletion:(void (^)(NSString *path, NSError *error))completion {
   405	    [_lock lock];
